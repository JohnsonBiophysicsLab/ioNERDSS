def icos_face_leg_reduce(COM: float, leg: float, sigma: float):
    n = 12
    angle = math.acos(-5**0.5/3)
    red_len = sigma/(2*math.sin(angle/2))
    ratio = 1 - red_len/distance(COM, leg)
    leg_red = []
    for i in range(0, 3):
        leg_red.append(round((leg[i] - COM[i])*ratio + COM[i], n))
    return leg_red

    v2 = [0, -1, m]

    v6 = [-1, m, 0]

    v8 = [m, 0, 1]
    v9 = [m, 0, -1]

    for i in VertCoord:
        temp_list = []

    return VertCoord_
